    "# Function to calculate the price of a bond\n",
    "def bond_price(y, C, F, T):\n",
    "    '''\n",
    "    y is the yield to maturity (as a decimal)\n",
    "    C is the annual coupon payment\n",
    "    F is the face value of the bond\n",
    "    T is the time to maturity (in years)\n",
    "\n",
    "    The output is the price of the bond defined as the present value of future cash flows.\n",
    "    This formula assumes:\n",
    "     - annual coupon payments\n",
    "     - flat yield curve\n",
    "\n",
    "    Note: one can extend this function to:\n",
    "     - handle different compounding frequencies\n",
    "     - handle non-flat term structures\n",
    "    ''' \n",
    "    \n",
    "    return sum([C / (1+y)**t for t in range(1, T+1)]) + F / (1+y)**T"

    "# Function to calculate the yield to maturity of a bond\n",
    "def ytm(price, C, F, T, guess=0.01, tol=1e-9):\n",
    "    '''\n",
    "    price is the current price of the bond\n",
    "    C is the annual coupon payment\n",
    "    F is the face value of the bond\n",
    "    T is the time to maturity (in years)\n",
    "    guess is an initial guess for the yield to maturity\n",
    "    tol is the tolerance level for convergence\n",
    "\n",
    "    The output is the yield to maturity (as a decimal) that solves the equation:\n",
    "    bond_price(y, C, F, T) = price\n",
    "\n",
    "    This function assumes:\n",
    "        - annual coupon payments \n",
    "    \n",
    "    Note 1: Even if the term structure is not flat, this function still computes\n",
    "    the yield to maturity, which is a single rate that equates the bond's price\n",
    "    to the present value of its cash flows.\n",
    "        \n",
    "    Note 2: The newton method iteratively adjusts y (the yield to maturity) until \n",
    "    the difference between the theoretical bond price and the observed price \n",
    "    is zero (or close to zero within the tolerance level).\n",
    "    '''\n",
    "    func = lambda y: bond_price(y, C, F, T) - price\n",
    "    return newton(func, guess, tol=tol)\n"
   ]
